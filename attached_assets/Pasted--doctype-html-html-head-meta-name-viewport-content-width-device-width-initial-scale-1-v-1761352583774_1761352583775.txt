<script>
/*
Robusto: atualiza --app-height com visualViewport quando disponível,
aplica transform no inputWrapper para "levantar" sobre o teclado e
faz scroll automático para o final quando o input foca.
*/

const root = document.documentElement;
const chatContainer = document.getElementById('chatContainer');
const messages = document.getElementById('messages');
const chatInput = document.getElementById('chatInput');
const inputWrapper = document.getElementById('inputWrapper');

function setAppHeight(px) {
  root.style.setProperty('--app-height', (px) + 'px');
}

/* fallback inicial */
setAppHeight(window.innerHeight);

/* função que rola para baixo */
function scrollToBottom(immediate = false) {
  const last = messages.lastElementChild;
  if (last) {
    // scrollIntoView é mais confiável que scrollTop para mobile
    last.scrollIntoView({ behavior: immediate ? 'auto' : 'smooth', block: 'end' });
  } else {
    messages.scrollTop = messages.scrollHeight;
  }
}

/* quando usar visualViewport (melhor no mobile) */
if (window.visualViewport) {
  const vv = window.visualViewport;

  function onViewportChange() {
    // vv.height é a área visível - exclui teclado
    const visibleH = Math.round(vv.height);
    // atualiza altura do app para visualizar corretamente
    setAppHeight(visibleH);

    // calcular teclado aproximado
    const keyboardH = Math.round(window.innerHeight - visibleH);

    // se teclado aberto (keyboardH > 0), levante o inputWrapper
    if (keyboardH > 0) {
      // desloca o input para cima pelo valor do teclado (suave)
      inputWrapper.style.transform = `translateY(-${keyboardH}px)`;
      // garante padding bottom nas mensagens para não ficar escondido
      messages.style.paddingBottom = (keyboardH + inputWrapper.offsetHeight + 12) + 'px';
      // rola para o final
      scrollToBottom();
    } else {
      // teclado fechado
      inputWrapper.style.transform = '';
      messages.style.paddingBottom = '';
    }
  }

  vv.addEventListener('resize', onViewportChange);
  vv.addEventListener('scroll', onViewportChange);
  window.addEventListener('orientationchange', ()=> setTimeout(()=> setAppHeight(window.innerHeight), 300));
  // inicial
  onViewportChange();
} else {
  // fallback: resize e focusin/out
  window.addEventListener('resize', () => {
    setTimeout(()=> setAppHeight(window.innerHeight), 50);
  });
}

/* focus handlers: rolar e ajustar (fallback útil) */
chatInput.addEventListener('focus', () => {
  // Em muitos devices, foco ativa teclado e resize/visualViewport cuida do resto.
  // Forçar pequeno delay e scroll
  setTimeout(()=> {
    scrollToBottom();
    // também tenta definir a altura real caso resize não ocorra
    setAppHeight(window.innerHeight);
  }, 300);
});

chatInput.addEventListener('blur', () => {
  // remove transform depois do teclado fechar
  setTimeout(()=> {
    inputWrapper.style.transform = '';
    messages.style.paddingBottom = '';
    setAppHeight(window.innerHeight);
  }, 300);
});

/* Enviar mensagem (exemplo) */
document.getElementById('sendBtn').addEventListener('click', () => {
  const v = chatInput.value.trim();
  if (!v) return;
  const div = document.createElement('div');
  div.className = 'message';
  div.textContent = v;
  messages.appendChild(div);
  chatInput.value = '';
  scrollToBottom();
});

/* Ao carregar, garanta scroll para o fim */
window.addEventListener('load', () => {
  setTimeout(()=>{ setAppHeight(window.innerHeight); scrollToBottom(true); }, 200);
});

/* Se estiver em iframe (preview do Replit), mostre console warning para o dev */
if (window.self !== window.top) {
  console.warn('Aviso: você está em iframe (preview do Replit). Abra o link público em nova aba para testes reais em mobile.');
}
</script>
